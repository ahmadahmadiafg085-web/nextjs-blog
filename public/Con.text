    <div id="log"></div>

    <script>
        const log = msg => {
            logBox.textContent += msg + "\n";
        };

        const logBox = document.getElementById("log");

        async function loadContainer() {
            try {
                const loader = await fetch("/loader").then(r => r.json());
                log("> Loader:");
                log(JSON.stringify(loader, null, 2));

                const container = await fetch(loader.url).then(r => r.text());
                log("\n> Container:");
                log(container);

            } catch (err) {
                log("> Error: " + err.message);
            }
        }

        document.getElementById("load").onclick = loadContainer;
    </script>
